var postorderTraversal = function (root) {
  var res = [];
  if (root === null) {
    return res;
  }

  var stack = [];
  stack.push(root);
  while (stack.length) {
    var curr = stack.pop();
    if (curr.left) {
      stack.push(curr.left);
    }
    if (curr.right) {
      stack.push(curr.right);
    }
    res.unshift(curr.val);
  }

  return res;
};

## 解题代码

```js
/**
 * @param {number[]} nums1
 * @param {number} m
 * @param {number[]} nums2
 * @param {number} n
 * @return {void} Do not return anything, modify nums1 in-place instead.
 */
var merge = function (nums1, m, nums2, n) {
  const len = m + n;
  for (let i = len - 1; i >= 0; i--) {
    let num1 = nums1[m - 1];
    let num2 = nums2[n - 1];

    if (n == 0 || (m > 0 && n > 0 && num1 > num2)) {
      nums1[i] = num1;
      m--;
    } else {
      nums1[i] = num2;
      n--;
    }
  }
};
```

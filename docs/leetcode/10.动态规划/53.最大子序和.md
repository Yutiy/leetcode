## 解题代码

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function (nums) {
  let pre = 0;
  let result = nums[0];

  nums.forEach((num) => {
    pre = Math.max(pre + num, num);
    result = Math.max(result, pre);
  });

  return result;
};
```

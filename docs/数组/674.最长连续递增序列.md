## 解题代码

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var findLengthOfLCIS = function (nums) {
  let len = nums.length;
  if (len === 0) return len;

  let res = 1,
    max = res;
  for (let i = 1; i < len - 1; i++) {
    if (nums[i] > nums[i - 1]) {
      res++;
    } else {
      res = 1;
    }
    max = Math.max(max, res);
  }

  return res;
};
```

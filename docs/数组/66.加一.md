## 解题代码

```js
/**
 * @param {number[]} digits
 * @return {number[]}
 */
var plusOne = function (digits) {
  let len = digits.length - 1;

  for (let i = len; i >= 0; i--) {
    let num = digits[i] + 1;
    if (num < 10) {
      digits[i] = num;
      return digits;
    } else {
      digits[i] = 0;
    }
  }

  return !digits[0] ? [1, ...digits] : digits;
};
```

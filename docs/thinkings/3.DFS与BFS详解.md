## 深度优先遍历 DFS

用到了栈结构，先进后出。

### 递归写法

```js
function dfs(node, nodeList = []) {
  if (node !== null) {
    nodeList.push(node);
    for (let i = 0, childs = node.children; i < childs.length; i++) {
      dfs(childs[i]);
    }
  }
  return nodeList;
}
```

### 非递归写法

```js
function dfs(root) {
  let res = [];

  if (root !== null) {
    let stack = [];
    stack.push(root);

    while (stack) {
      const curr = stack.pop();
      res.push(curr);

      let childs = curr.children;
      for (let i = childs.length - 1; i >= 0; i--) {
        stack.push(childs[i]);
      }
    }
  }
  return res;
}
```

## 解题代码

```js
/**
 * @param {number} x
 * @return {boolean}
 */
var isPalindrome = function (x) {
  if (x < 0 || (x % 10 == 0 && x != 0)) return false;
  return `${x}`.split("").reverse().join("") === `${x}`;
};
```

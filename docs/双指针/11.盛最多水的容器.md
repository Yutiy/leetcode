## 解题代码

```js
/**
 * @param {number[]} height
 * @return {number}
 */
var maxArea = function (height) {
  let l = 0;
  let r = height.length - 1;

  let res = 0;
  while (l < r) {
    let value = (r - l) * Math.min(height[l], height[r]);
    if (height[l] > height[r]) {
      r--;
    } else {
      l++;
    }

    res = Math.max(value, res);
  }

  return res;
};
```
